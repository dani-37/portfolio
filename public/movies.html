def plot_counts_cdf(df):
    # Average ratings per movie
    _, movie_counts = np.unique(df['movieId'], return_counts=True)
    movie_hist, movie_bin_edges = np.histogram(movie_counts, bins=10000, density=True)
    movie_cdf = np.cumsum(movie_hist) * np.diff(movie_bin_edges)
    movie_stats = (np.min(movie_counts), np.max(movie_counts), np.mean(movie_counts), np.std(movie_counts))

    # Average ratings per user
    _, user_counts = np.unique(df['userId'], return_counts=True)
    user_hist, user_bin_edges = np.histogram(user_counts, bins=10000, density=True)
    user_cdf = np.cumsum(user_hist) * np.diff(user_bin_edges)
    user_stats = (np.min(user_counts), np.max(user_counts), np.mean(user_counts), np.std(user_counts))

    # Plot side by side
    _, axes = plt.subplots(1, 2, figsize=(15, 6))

    # Cumulative histogram of average ratings per user
    axes[0].plot(user_bin_edges[:-1], user_cdf, marker='o', linestyle='-', color='skyblue', 
                 label=(f"Min:    {user_stats[0]:.0f}\nMax:   {user_stats[1]:,.0f}\n"
                        f"Mean: {user_stats[2]:,.2f}\nStd:     {user_stats[3]:,.2f}"))
    axes[0].set_title('Cumulative Histogram of Average Number of Ratings per User')
    axes[0].set_xlabel('Number of Ratings')
    axes[0].set_ylabel('Cumulative Frequency')
    axes[0].set_xscale('log')
    axes[0].legend(loc="lower right")

    # Cumulative histogram of average ratings per movie
    axes[1].plot(movie_bin_edges[1:], movie_cdf, marker='o', linestyle='-', color='orange',
                 label=(f"Min:    {movie_stats[0]:.0f}\nMax:   {movie_stats[1]:,.0f}\n"
                        f"Mean: {movie_stats[2]:,.2f}\nStd:     {movie_stats[3]:,.2f}"))
    axes[1].set_title('Cumulative Histogram of Average Number of Ratings per Movie')
    axes[1].set_xlabel('Number of Ratings')
    axes[1].set_ylabel('Cumulative Frequency')
    axes[1].set_xscale('log')
    axes[1].legend(loc="lower right")

    plt.tight_layout()
    plt.show()

# Calculate common counts and confidence
confidence_base = 3.0
mask = coo_matrix((np.ones_like(R.data, dtype=float), (R.row, R.col)), shape=R.shape)
common_count = mask @ mask.T  # MxM result
confidence = common_count.copy()
confidence.data = confidence.data / (confidence.data + confidence_base)

# Numerator: (R*S)*R.T
numerator = (S.multiply(R) @ R.T).toarray()

# Compute squares
RR = R.power(2)
SS_RR = S.power(2).multiply(RR)

# Norm calculations
norm1_sq = (RR @ mask.T).toarray()
norm2_sq = (SS_RR @ mask.T).toarray()

# Use epsilon to reduce norm impact
epsilon = 1.75
norm1 = np.sqrt(norm1_sq) + epsilon
norm2 = np.sqrt(norm2_sq) + epsilon
denominator = norm1*norm2

# Compute edges
similarity = np.divide(numerator, denominator, out=np.zeros_like(numerator), where=(denominator)!=0)
edges = similarity * confidence.toarray()

def find_movie_ids(movie_list: list):
    movie_ids = []

    for title in movie_list:
        # Normalize title for "The" handling
        normalized_title = title.lower()
        if normalized_title.startswith("the "): normalized_title = normalized_title[4:] + ", the"

        # First check for exact match
        exact_match = movies[movies['title'].str.lower() == normalized_title]
        if not exact_match.empty:
            movie_ids.append(exact_match.index[0])
            continue 
        
        # If no exact match, use contains (fallback)
        partial_match = movies[movies['title'].str.lower().str.contains(title.lower())]
        if not partial_match.empty: movie_ids.append(partial_match.index[0])
    
    return movie_ids


def find_movies(movie_list: list, top_n: int = 10):
    movie_ids = find_movie_ids(movie_list)

    missing_movies = [(movie_list[i], movie_ids[i]) 
                      for i in range(len(movie_ids)) 
                      if movie_ids[i] not in movie_to_index]
    
    assert not missing_movies, f"The following movies are missing in 'movie_to_index': {missing_movies}"
    
    input_indices = [movie_to_index[m] for m in movie_ids]    
    candidate_indices = [i for i in range(len(unique_movie_ids)) if unique_movie_ids[i] not in movie_ids]

    scores = []
    for c_idx in candidate_indices:
        weights_from_input = [edges[i_idx, c_idx] for i_idx in input_indices]
        combined_score = np.mean(weights_from_input)  # Simple average for clarity
        if combined_score > 0: scores.append((c_idx, combined_score))
    
    scores.sort(key=lambda x: x[1], reverse=True)
    top_candidates = scores[:top_n]

    # Create a list of dictionaries to build the DataFrame
    results = [
        {
            'title': movies.loc[unique_movie_ids[c_idx], 'title'] 
                    if unique_movie_ids[c_idx] in movies.index 
                    else f"Unknown Movie ID {unique_movie_ids[c_idx]}",
            'score': int(100 * score)
        }
        for c_idx, score in top_candidates
    ]

    print(movies.loc[movie_ids])
    display(pd.DataFrame(results))
